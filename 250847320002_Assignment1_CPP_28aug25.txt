/*
12. Write a program calculate the sum of fibonacci series.
*/
#include <iostream>
using namespace std;
int main()
{
    int n,t1,t2,sum;
    t1 = 0, t2 = 1;
	sum = t1 + t2;
    cout << "Enter the number of terms in the Fibonacci series: ";
    cin >> n;
    if (n <= 0) {
        cout << "Please enter a positive integer." << endl;
        return 1;
    } else if (n == 1) {
        cout << "Fibonacci Series: " << t1 << endl;
        cout << "Sum: " << t1 << endl;
    } else {
        cout << "Fibonacci Series: " << t1 << ", " << t2;
        for (int i = 2; i < n; ++i)
        {
            int nextTerm = t1 + t2;
            sum += nextTerm;
            cout << ", " << nextTerm;
            t1 = t2;
            t2 = nextTerm;
        }
        cout << endl;
        cout << "Sum of the series: " << sum << endl;
    }
    return 0;
}
